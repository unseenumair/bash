2. Production Ready Pattern


using
  case STRING in
    PATTERN1) COMMANDS ;;
    PATTERN2) COMMANDS ;;
    *) COMMANDS ;;  # default
  esac

as
  case "${input,,}" in
    yes|y) echo "Proceed" ;;
    no|n)  echo "Abort" ;;
    *)     echo "Invalid" ;;
  esac
